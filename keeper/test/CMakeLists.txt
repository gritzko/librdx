project(git_test C)
set(CMAKE_C_STANDARD 23)
include(CTest)
include_directories(.. ../..)

set(TEST_CXX_FLAGS "-g")

add_executable(GITtest GIT.c)
target_link_options(GITtest PRIVATE ${TEST_LDD_FLAGS})
